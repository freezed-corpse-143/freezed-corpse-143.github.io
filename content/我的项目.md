# 导出当前文件夹及其递归子文件夹下所有py文件中import的库并保存

将以下代码，保存到import_modules.py，移动目标文件夹下，并运行

```python
import os
import re

def extract_imports(file_path):
    with open(file_path, 'r', encoding='utf-8') as file:
        content = file.read()
        # 使用正则表达式匹配 import 语句
        imports = re.findall(r'import (\w+)|from (\w+) import', content)
        # 将匹配结果展平并去重
        return set(item for sublist in imports for item in sublist if item)

def find_all_py_files(directory):
    py_files = []
    for root, _, files in os.walk(directory):
        for file in files:
            if file.endswith('.py'):
                py_files.append(os.path.join(root, file))
    return py_files

def main():
    current_directory = os.getcwd()
    py_files = find_all_py_files(current_directory)
    all_imports = set()

    for py_file in py_files:
        all_imports.update(extract_imports(py_file))

    # 将结果保存到 txt 文件
    with open('imports.txt', 'w', encoding='utf-8') as output_file:
        for import_name in sorted(all_imports):
            output_file.write(f"{import_name}\n")

if __name__ == "__main__":
    main()
```

# 时间测量装饰器

```python
import logging

logging.basicConfig(format = '%(asctime)s - %(levelname)s - %(name)s -   %(message)s',
                    datefmt = '%m/%d/%Y %H:%M:%S',
                    level = logging.INFO)
logger = logging.getLogger(__name__)

def timed(func):
    """This decorator prints the execution time for the decorated function."""

    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        logger.debug("{} ran in {}s".format(func.__name__, round(end - start, 2)))
        return result
    return wrapper

@timed
def main():
    pass
```

# 批量修改文件夹下文件后缀

```python
import os
import shutil

def rename_files(folder_path, old_suffix, new_suffix):
    # 遍历文件夹下的所有文件
    for filename in os.listdir(folder_path):
        # 检查文件是否以旧后缀结尾
        if filename.endswith(old_suffix):
            # 构建新的文件名，将旧后缀替换为新后缀
            new_filename = filename.replace(old_suffix, new_suffix)
            
            # 构建完整的源文件路径和目标文件路径
            source_file = os.path.join(folder_path, filename)
            target_file = os.path.join(folder_path, new_filename)
            
            # 重命名文件
            shutil.move(source_file, target_file)
            print(f'Renamed {filename} to {new_filename}')

    print(f"All files with suffix '{old_suffix}' have been renamed to '{new_suffix}'.")
```

```javascript
def print_tree(node, prefix="", is_last=True):
    # 打印当前节点
    print(prefix + ("└── " if is_last else "├── ") + str(node))

    # 更新前缀
    if is_last:
        prefix += "    "
    else:
        prefix += "│   "

    # 递归打印子节点
    for i, child in enumerate(node.children):
```

```python
import openpyxl
def get_cells_info(sheet):
    result = []
    merged_ranges = [merged_range.bounds for merged_range in sheet.merged_cells.ranges]
    for merged_range in sheet.merged_cells.ranges:
        output = ""
        min_col, min_row, max_col, max_row = merged_range.bounds
        
        content_cell = sheet.cell(row=min_row, column=min_col)
        content = content_cell.value if content_cell.value is not None else "无内容"
        output += f"内容: {content}\t"
        output += "包含单元格: "
        
        for row in range(min_row, max_row + 1):
            for col in range(min_col, max_col + 1):
                cell_address = f"第{str(row)}行 第{str(col)}列"
                output += cell_address + " "
        result.append(output)
    for row in range(1, sheet.max_row + 1):
        for col in range(1, sheet.max_column + 1):
            cell = sheet.cell(row=row, column=col)
            
            in_merged = False
            for min_col, min_row, max_col, max_row in merged_ranges:
                if min_col <= col <= max_col and min_row <= row <= max_row:
                    in_merged = True
                    break
            
            if not in_merged and cell.value is not None:
                output = f"内容: {cell.value}\t"
                output += f"单元格位置: 第{str(row)}行，第{str(col)}列"
                result.append(output)
    
    return "\n".join(result)
```

# 快速读取大文件中的行

> 使用内存索引

```python
class TextIndex:
    def __init__(self, txt_path):
        import mmap
        with open(txt_path, 'r+') as f:
            mm = mmap.mmap(f.fileno(), 0, access=mmap.ACCESS_READ)
            index = []
            pos = 0
            while True:
                index.append(pos)
                pos = mm.find(b'\n', pos) + 1
                if pos == 0:
                    break
        self.mm = mm
        self.index = index
        self.index_len = len(index)
    def id2doc(self, id):
        if id < 0 or id >= self.index_len - 1:
            return None
        start = self.index[id]
        end = self.index[id + 1] if id + 1 < self.index_len else self.mm.size()
        return self.mm[start:end].decode('utf-8')
    
    def __del__(self):
        if hasattr(self, 'mm'):
            self.mm.close()
        if hasattr(self, 'index'):
            self.index = []
```

6. 批量爬取脚本，`batch_crawl.sh`

```bash
#!/bin/bash

# 设置脚本所在目录为工作目录
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
cd "$SCRIPT_DIR"

# 定义状态文件，用于记录已处理的URL
PROCESSED_LOG=".processed_urls.log"

# 颜色输出函数
print_green() {
    echo -e "\033[32m$1\033[0m"
}

print_yellow() {
    echo -e "\033[33m$1\033[0m"
}

print_red() {
    echo -e "\033[31m$1\033[0m"
}

# 显示使用说明
show_usage() {
    echo "使用方法: $0 [URL文件路径]"
    echo "  URL文件路径: 包含网站链接的文件，每行一个链接"
    echo "  如果不指定，默认使用当前目录下的 urls.txt"
    echo ""
    echo "示例:"
    echo "  $0                     # 使用默认的 urls.txt"
    echo "  $0 my_links.txt        # 使用指定的 my_links.txt"
    echo "  $0 /path/to/urls.txt   # 使用绝对路径的URL文件"
}

# 获取URL文件路径（第一个参数，默认为urls.txt）
URLS_FILE="${1:-urls.txt}"

# 检查是否请求帮助
if [ "$1" = "-h" ] || [ "$1" = "--help" ]; then
    show_usage
    exit 0
fi

# 检查crawl.sh是否存在
if [ ! -f "crawl.sh" ]; then
    print_red "错误: crawl.sh 文件不存在于当前目录！"
    exit 1
fi

# 确保crawl.sh有执行权限
chmod +x crawl.sh

# 检查URL文件是否存在
if [ ! -f "$URLS_FILE" ]; then
    print_red "错误: URL文件 '$URLS_FILE' 不存在！"
    echo ""
    show_usage
    exit 1
fi

# 创建状态文件（如果不存在）
touch "$PROCESSED_LOG"

# 初始化计数器
total_urls=$(grep -c '[^[:space:]]' "$URLS_FILE" 2>/dev/null || echo "0")
if [ "$total_urls" -eq 0 ]; then
    print_yellow "警告: URL文件 '$URLS_FILE' 是空的或只包含空白行！"
    exit 0
fi

processed_count=0
skipped_count=0
failed_count=0

print_yellow "使用URL文件: $URLS_FILE"
print_yellow "开始处理URLs，总共 $total_urls 个链接..."
echo "----------------------------------------"

# 逐行读取URLs文件
while IFS= read -r url || [ -n "$url" ]; do
    # 跳过空行和只包含空白字符的行
    if [[ -z "${url// }" ]]; then
        continue
    fi
    
    # 去除URL两端的空白字符
    url=$(echo "$url" | xargs)
    
    # 检查URL是否已经处理过（幂等性检查）
    if grep -Fxq "$url" "$PROCESSED_LOG"; then
        print_yellow "跳过已处理的URL: $url"
        ((skipped_count++))
        continue
    fi
    
    print_green "处理URL: $url"
    
    # 执行crawl.sh脚本
    if bash crawl.sh "$url"; then
        # 执行成功，记录到状态文件
        echo "$url" >> "$PROCESSED_LOG"
        print_green "✓ 成功处理: $url"
        ((processed_count++))
    else
        print_red "✗ 处理失败: $url"
        ((failed_count++))
    fi
    
    # 检查是否还有更多URL需要处理
    if [ $((processed_count + skipped_count + failed_count)) -lt $total_urls ]; then
        # 生成随机等待时间（5-10秒）
        sleep_time=$(( RANDOM % 6 + 5 ))
        print_yellow "等待 ${sleep_time} 秒后继续下一个请求..."
        sleep $sleep_time
    fi
    
done < "$URLS_FILE"

echo "----------------------------------------"
print_green "处理完成！"
echo "URL文件: $URLS_FILE"
echo "总计: $total_urls 个URL"
echo "成功: $processed_count 个"
echo "跳过: $processed_count 个"
echo "失败: $failed_count 个"

# 显示已处理的URL列表
if [ -s "$PROCESSED_LOG" ]; then
    echo ""
    print_yellow "已处理的URL列表保存在: $PROCESSED_LOG"
fi
```

```python
    统一的HTTP处理器：
    - GET: 支持 Range 头（多线程下载）和 sendfile 零拷贝。
    - PUT: 支持 Content-Range 头（多线程上传）和 atomic pwrite。
    """
    
    server_version = "AutoDL-Unified/3.0"
    
    def __init__(self, *args, **kwargs):
        self.logger = self._setup_logging()
        super().__init__(*args, **kwargs)
    
    def _setup_logging(self):
        logger = logging.getLogger(__name__)
        logger.setLevel(logging.INFO)
        handler = logging.StreamHandler()
        handler.setFormatter(logging.Formatter('%(asctime)s - %(levelname)s - %(message)s'))
        logger.addHandler(handler)
        return logger
    
    def log_message(self, format: str, *args) -> None:
        self.logger.info("%s - - [%s] %s\n" % (
            self.address_string(),
            self.log_date_time_string(),
            format % args))

    # ============================
    # 路由与上传逻辑 (PUT)
    # ============================

    def do_GET(self):
        """路由分发：根路径返回上传面板，其他路径作为文件服务器"""
        if self.path == '/' or self.path == '/upload':
            self.send_response(HTTPStatus.OK)
            self.send_header("Content-type", "text/html; charset=utf-8")
            self.send_header("Content-Length", str(len(UPLOAD_UI_HTML.encode('utf-8'))))
            self.end_headers()
            self.wfile.write(UPLOAD_UI_HTML.encode('utf-8'))
        else:
            # 这是一个小优化：如果用户访问 /files/，我们列出目录
            # 如果是具体文件，SimpleHTTPRequestHandler 会调用 send_head
            super().do_GET()

    def do_PUT(self):
        """处理并发上传分片"""
        try:
            filename = os.path.basename(self.path)
            from urllib.parse import unquote
            filename = unquote(filename)
            filepath = os.path.join(os.getcwd(), filename)

            content_range = self.headers.get('Content-Range')
            content_length = int(self.headers.get('Content-Length', 0))

            if not content_range:
                self.send_error(HTTPStatus.BAD_REQUEST, "Missing Content-Range header")
                return

            match = re.search(r'bytes (\d+)-(\d+)/(\d+)', content_range)
            if not match:
                self.send_error(HTTPStatus.BAD_REQUEST, "Invalid Content-Range format")
                return

            start_offset = int(match.group(1))
            
            # 读取数据
            data_buffer = self.rfile.read(content_length)

            # 原子写入：使用 pwrite 避免 seek 竞争
            fd = os.open(filepath, os.O_CREAT | os.O_WRONLY, 0o664)
            try:
                os.pwrite(fd, data_buffer, start_offset)
            finally:
                os.close(fd)

            self.send_response(HTTPStatus.OK)
            self.send_header('Access-Control-Allow-Origin', '*')
            self.end_headers()

        except Exception as e:
            self.log_error(f"Upload error: {e}")
            self.send_error(HTTPStatus.INTERNAL_SERVER_ERROR, str(e))

    def do_OPTIONS(self):
        self.send_response(HTTPStatus.OK)
        self.send_header('Access-Control-Allow-Origin', '*')
        self.send_header('Access-Control-Allow-Methods', 'PUT, GET, OPTIONS')
        self.send_header('Access-Control-Allow-Headers', 'Content-Type, Content-Range')
        self.end_headers()

    # ============================
    # 下载优化逻辑 (GET / sendfile)
    # ============================

    def send_head(self):
        """
        重写 send_head 以支持：
        1. Range 请求 (断点续传/多线程下载)
        2. 206 Partial Content 响应
        """
        path = self.translate_path(self.path)
        
        # 目录处理
        if os.path.isdir(path):
            return super().send_head()

        if not os.path.exists(path):
            self.send_error(HTTPStatus.NOT_FOUND, "File not found")
            return None
        
        f = None
        try:
            f = open(path, 'rb')
        except OSError:
            self.send_error(HTTPStatus.NOT_FOUND, "File not found")
            return None
        
        fs = os.fstat(f.fileno())
        file_size = fs.st_size
        start_range = 0
        end_range = file_size - 1
        
        # 处理 Range 头
        if 'Range' in self.headers:
            try:
                range_header = self.headers['Range']
                if not range_header.startswith('bytes='):
                    raise ValueError("Invalid range unit")
                
                ranges = range_header[6:].split('-')
                if ranges[0]: start_range = int(ranges[0])
                if ranges[1]: end_range = int(ranges[1])
                
                if start_range >= file_size:
                    self.send_error(HTTPStatus.REQUESTED_RANGE_NOT_SATISFIABLE, "Range not satisfiable")
                    return None
                
                self.send_response(HTTPStatus.PARTIAL_CONTENT)
                self.send_header('Content-Range', f'bytes {start_range}-{end_range}/{file_size}')
                content_length = end_range - start_range + 1
            except ValueError:
                self.send_error(HTTPStatus.BAD_REQUEST, "Invalid Range header")
                return None
        else:
            self.send_response(HTTPStatus.OK)
            content_length = file_size
        
        self.send_header('Accept-Ranges', 'bytes')
        self.send_header('Content-Length', str(content_length))
        self.send_header('Last-Modified', self.date_time_string(fs.st_mtime))
        
        ctype = self.guess_type(path)
        if ctype: self.send_header('Content-Type', ctype)
        self.end_headers()
        
        if self.command == 'HEAD':
            f.close()
            return None
            
        # 核心优化：使用 sendfile (Linux Kernel 零拷贝)
        if hasattr(os, 'sendfile') and hasattr(self.request, 'sendfile'):
            try:
                # sendfile(out_fd, in_fd, offset, count)
                # 注意：os.sendfile 的 offset 参数在 Python 中是 pointer 还是 int 取决于版本
                # 但 Python 的 socket.sendfile 更高级，处理了循环发送
                # 这里我们用 simpler os.sendfile or socket.sendfile
                # 为了兼容 Range，我们使用 socket.sendfile (如果可用) 或者 os.sendfile
                self.request.sendfile(f, start_range, content_length)
                f.close()
                return None # 返回 None 告诉 do_GET 不要再写数据了
            except (OSError, AttributeError):
                pass 
        
        # 降级方案：普通 seek + read
        f.seek(start_range)
        return f

    def copyfile(self, source, outputfile):
        """
        重写 copyfile：如果 do_GET 拿到的是普通文件对象（send_head 返回了 f），
        再次尝试使用 sendfile 进行全量复制（针对非 Range 请求的优化）
        """
        try:
            if hasattr(os, 'sendfile') and hasattr(outputfile, 'sendfile'):
                # 针对全文件下载的零拷贝
                os.sendfile(outputfile.fileno(), source.fileno(), None, os.fstat(source.fileno()).st_size)
                return
        except (OSError, AttributeError):
            pass
        super().copyfile(source, outputfile)
```

```powershell
    $remote_commit = git rev-parse origin/$branch 2>$null

    # Check for uncommitted local changes
    $status = git status --porcelain

    # Only pull when remote actually has updates
    if ($remote_commit -and $local_commit -ne $remote_commit)
    {
        Write-Host "Remote is ahead, pulling changes..." -ForegroundColor Cyan

        # Stash if there are local changes
        $hasLocalChanges = -not [string]::IsNullOrWhiteSpace($status)
        if ($hasLocalChanges)
        {
            Write-Host "Local changes detected, stashing..." -ForegroundColor Yellow
            git stash push -m "Auto-stash before pull $(Get-Date)"
        }

        # Execute pull
        git pull origin $branch --quiet

        # Try to apply stash if exists
        $stashList = git stash list
        if ($stashList -match "Auto-stash before pull")
        {
            Write-Host "Applying stashed changes..." -ForegroundColor Yellow

            git stash apply --quiet 2>$null
            if ($LASTEXITCODE -eq 0)
            {
                git stash drop --quiet 2>$null
            }

            # Recheck status, as stash pop may create new files or conflicts
            $status = git status --porcelain
        }
    }

    # Check for local changes and commit - ensure not empty
    if (-not [string]::IsNullOrWhiteSpace($status))
    {
        Write-Host "Changes detected: $status" -ForegroundColor Yellow
        git add .
        git commit -m "Auto: $(Get-Date)"

        # Check remote status again to prevent new commits during local commit process
        git fetch origin $branch --quiet 2>$null
        $local_commit = git rev-parse $branch
        $remote_commit = git rev-parse origin/$branch 2>$null

        if ($remote_commit -and $local_commit -ne $remote_commit)
        {
            Write-Host "Remote changed during commit, pulling and rebasing..." -ForegroundColor Cyan
            git pull --rebase origin $branch --quiet
        }

        # Push commits
        $pushResult = git push origin $branch 2>&1
        if ($LASTEXITCODE -eq 0)
        {
            Write-Host "Changes committed and pushed at $(Get-Date)" -ForegroundColor Green
            # optional: display push result but not as error
            Write-Host $pushResult -ForegroundColor Gray
        } else
        {
            Write-Host "Push failed: $pushResult" -ForegroundColor Red
        }

        # Key fix: reset status variable to empty
        $status = $null

        # Additional fix: wait a bit to avoid immediate re-detection
        Start-Sleep -Seconds 2
    }

    Start-Sleep -Seconds 5
}

Write-Host "Script stopped" -ForegroundColor Green
```

# 文件树重建目录

```bash
#!/bin/bash

# Check if tree.txt file is provided
if [ $# -ne 1 ]; then
    echo "Usage: $0 <tree.txt>"
    exit 1
fi

TREE_FILE="$1"

# Check if file exists
if [ ! -f "$TREE_FILE" ]; then
    echo "Error: File '$TREE_FILE' does not exist"
    exit 1
fi

# Get the directory containing tree.txt and switch to it
TREE_DIR=$(dirname "$TREE_FILE")
cd "$TREE_DIR" || exit 1

echo "Rebuilding directory structure from $TREE_FILE..."

# Temporary file to store filtered content
TEMP_FILE=$(mktemp)

# Remove comment lines (starting with "# ") and empty lines
grep -v '^# ' "$TREE_FILE" | grep -v '^$' > "$TEMP_FILE"

# Find the first non-blank line starting with ├ or └ or │
START_LINE=$(grep -n -m 1 '^[ │├└]' "$TEMP_FILE" | cut -d: -f1)

if [ -z "$START_LINE" ]; then
    echo "Error: No valid directory tree structure found"
    rm -f "$TEMP_FILE"
    exit 1
fi

# Extract content from the start line to the end
tail -n +"$START_LINE" "$TEMP_FILE" > "$TEMP_FILE.tree"
mv "$TEMP_FILE.tree" "$TEMP_FILE"

# Store all lines, depths, and names
declare -a lines
declare -a depths
declare -a names

# First pass: parse all lines, calculate depth and name
while IFS= read -r line; do
    # Skip empty lines
    [ -z "$line" ] && continue

    # Calculate depth: count leading "│   " or "    " occurrences
    # Extract leading tree symbols
    prefix=$(echo "$line" | sed -E 's/^([ │├└─]+).*$/\1/')

    # Calculate depth: each "│   " counts as one level, "│       " counts as two levels (since it's shorthand for "│   │   ")
    depth=0
    # Normalize: replace "│       " with "||" (two levels), "│   " with "|" (one level), "    " with " "
    normalized=$(echo "$prefix" | sed 's/│       /||/g; s/│   /|/g; s/    / /g')
    depth=$(echo "$normalized" | grep -o '|' | wc -l)

    # Extract name
    name=$(echo "$line" | sed -E 's/^[ │├└─]+//; s/^[ │├└]+//; s/^[ ]*//')
    # Remove comments
    name=$(echo "$name" | sed 's/[[:space:]]*#.*$//')
    # Remove trailing spaces
    name=$(echo "$name" | sed 's/[[:space:]]*$//')

    # Skip empty names
    [ -z "$name" ] && continue

    lines+=("$line")
    depths+=($depth)
    names+=("$name")

done < "$TEMP_FILE"

# Second pass: determine whether each item is a directory or file
# Rules:
# 1. If the name ends with "/", it's a directory
# 2. Otherwise, if an item has deeper indented items after it, it's a directory, otherwise it's a file
declare -a is_dir
total_lines=${#names[@]}

for ((i=0; i<total_lines; i++)); do
    name=${names[$i]}
    current_depth=${depths[$i]}

    # Rule 1: Name ending with "/" is a directory
    if [[ "$name" == */ ]]; then
        is_dir+=("true")
        continue
    fi

    # Rule 2: Check if there are child items
    is_directory=false
    for ((j=i+1; j<total_lines; j++)); do
        next_depth=${depths[$j]}
        if [ $next_depth -gt $current_depth ]; then
            is_directory=true
            break
        elif [ $next_depth -eq $current_depth ]; then
            # Stop checking when encountering a sibling
            break
        fi
    done

    is_dir+=($is_directory)
done

# Third pass: create directories and files
# Use a stack to track current path
declare -a path_stack

for ((i=0; i<total_lines; i++)); do
    current_depth=${depths[$i]}
    name=${names[$i]}
    is_directory=${is_dir[$i]}

    # Adjust stack to current depth
    if [ $current_depth -lt ${#path_stack[@]} ]; then
        # Pop extra items
        while [ ${#path_stack[@]} -gt $current_depth ]; do
            unset path_stack[${#path_stack[@]}-1]
        done
    fi

    # Build full path
    if [ $current_depth -eq 0 ]; then
        full_path="$name"
        # Reset stack
        path_stack=()
        path_stack+=("$name")
    else
        # Build parent path
        parent_path=""
        for ((j=0; j<$current_depth; j++)); do
            if [ -z "$parent_path" ]; then
                parent_path="${path_stack[$j]}"
            else
                parent_path="$parent_path/${path_stack[$j]}"
            fi
        done
        full_path="$parent_path/$name"

        # Update stack
        if [ $current_depth -le ${#path_stack[@]} ]; then
            path_stack[$current_depth]="$name"
        else
            path_stack+=("$name")
        fi
    fi

    # Create file or directory
    if [ "$is_directory" = true ]; then
        # Create directory
        if [ ! -d "$full_path" ]; then
            mkdir -p "$full_path"
            echo "Creating directory: $full_path"
        else
            echo "Directory already exists, skipping: $full_path"
        fi
    else
        # Create file: ensure parent directory exists
        parent_dir=$(dirname "$full_path")
        if [ "$parent_dir" != "." ] && [ "$parent_dir" != "/" ]; then
            mkdir -p "$parent_dir"
        fi
        # Create empty file (if it doesn't exist)
        if [ ! -f "$full_path" ]; then
            touch "$full_path"
            echo "Creating file: $full_path"
        else
            echo "File already exists, skipping: $full_path"
        fi
    fi

done

# Clean up temporary file
rm -f "$TEMP_FILE"

echo "Rebuild completed!"
```

```bash
#!/bin/bash

# 用户管理脚本
# 功能: 列出用户、添加用户、删除用户

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# 默认密码
DEFAULT_PASSWORD="12345678"

# 日志函数
log_info() {
    echo -e "${GREEN}[INFO]${NC} $1"
}

log_warn() {
    echo -e "${YELLOW}[WARN]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

log_debug() {
    echo -e "${BLUE}[DEBUG]${NC} $1"
}

# 检查是否为 root 用户
check_root() {
    if [ "$EUID" -ne 0 ]; then 
        log_error "请使用 root 用户运行此脚本"
        exit 1
    fi
}

# 列出所有用户（包括系统用户）
list_users() {
    local show_all=${1:-false}
    
    echo -e "\n${BLUE}=== 用户列表 ===${NC}"
    
    if [ "$show_all" = "all" ] || [ "$show_all" = "-a" ]; then
        # 显示所有用户
        echo -e "${YELLOW}用户名\t\tUID\t\t家目录\t\tShell${NC}"
        echo "--------------------------------------------------------"
        awk -F: '{printf "%-20s %-10s %-20s %s\n", $1, $3, $6, $7}' /etc/passwd
    else
        # 显示普通用户（UID >= 1000 且 UID != 65534）和重要的服务用户
        echo -e "${YELLOW}用户名\t\tUID\t\t家目录\t\tShell\t\t类型${NC}"
        echo "--------------------------------------------------------"
        
        # 显示普通用户
        awk -F: 'BEGIN {count=0} $3 >= 1000 && $3 != 65534 {printf "%-20s %-10s %-20s %s\t%s\n", $1, $3, $6, $7, "[普通用户]"; count++} END {if(count==0) print "没有普通用户"}' /etc/passwd
        
        # 显示重要的服务用户（可选）
        echo -e "\n${BLUE}=== 重要系统服务用户 ===${NC}"
        for user in root postgres mysql www-data nginx redis mongodb; do
            if id "$user" &>/dev/null 2>&1; then
                user_info=$(getent passwd "$user")
                uid=$(echo "$user_info" | cut -d: -f3)
                home=$(echo "$user_info" | cut -d: -f6)
                shell=$(echo "$user_info" | cut -d: -f7)
                printf "%-20s %-10s %-20s %s\n" "$user" "$uid" "$home" "$shell"
            fi
        done
    fi
}

# 检查用户是否存在
user_exists() {
    id "$1" &>/dev/null 2>&1
    return $?
}

# 获取用户详细信息
get_user_info() {
    local username=$1
    if user_exists "$username"; then
        echo "用户名: $username"
        echo "UID: $(id -u "$username")"
        echo "GID: $(id -g "$username")"
        echo "家目录: $(eval echo ~$username)"
        echo "Shell: $(getent passwd "$username" | cut -d: -f7)"
        echo "组: $(id -Gn "$username" | tr ' ' ',')"
    fi
}

# 添加用户
add_user() {
    local username=$1
    local custom_password=$2
    
    # 检查用户名是否提供
    if [ -z "$username" ]; then
        log_error "请提供用户名"
        echo "用法: $0 add <用户名> [密码]"
        return 1
    fi
    
    # 验证用户名格式
    if [[ ! "$username" =~ ^[a-z_][a-z0-9_-]*$ ]]; then
        log_error "用户名格式不正确。只能包含小写字母、数字、下划线和连字符，且必须以字母或下划线开头"
        return 1
    fi
    
    # 检查用户是否已存在（幂等性）
    if user_exists "$username"; then
        log_warn "用户 '$username' 已经存在，跳过创建"
        echo -e "\n${GREEN}现有用户信息:${NC}"
        get_user_info "$username"
        return 0
    fi
    
    log_info "正在创建用户: $username"
    
    # 确定密码
    local password="${custom_password:-$DEFAULT_PASSWORD}"
    
    # 创建用户，自动创建家目录，指定 bash shell
    if useradd -m -s /bin/bash "$username" 2>/dev/null; then
        log_info "用户 '$username' 创建成功"
        
        # 设置密码
        echo "$username:$password" | chpasswd 2>/dev/null
        if [ $? -eq 0 ]; then
            log_info "密码已设置"
        else
            log_error "设置密码失败"
            return 1
        fi
        
        # 显示用户信息
        echo -e "\n${GREEN}✓ 用户创建成功！${NC}"
        echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
        echo "  用户名: $username"
        echo "  密码: $password"
        echo "  家目录: /home/$username"
        echo "  Shell: /bin/bash"
        echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
        
        # 可选：强制首次登录修改密码
        # passwd -e "$username" 2>/dev/null
        
    else
        log_error "创建用户 '$username' 失败"
        log_debug "请检查: useradd 命令是否可用，或用户名是否合法"
        return 1
    fi
}

# 删除用户
delete_user() {
    local username=$1
    local remove_home=${2:-false}
    
    # 检查用户名是否提供
    if [ -z "$username" ]; then
        log_error "请提供用户名"
        echo "用法: $0 del <用户名> [--remove-home]"
        return 1
    fi
    
    # 检查用户是否存在
    if ! user_exists "$username"; then
        log_error "用户 '$username' 不存在"
        return 1
    fi
    
    # 禁止删除重要系统用户
    local system_users="root daemon bin sys sync games man lp mail news uucp proxy www-data backup list irc gnats nobody systemd-timesync postgres mysql"
    if echo "$system_users" | grep -qw "$username"; then
        log_error "不允许删除重要系统用户: $username"
        return 1
    fi
    
    # 显示用户信息
    echo -e "\n${YELLOW}准备删除用户: $username${NC}"
    get_user_info "$username"
    echo
    
    # 检查是否有进程正在运行
    local user_pids=$(pgrep -u "$username" 2>/dev/null)
    if [ -n "$user_pids" ]; then
        log_warn "用户 '$username' 有正在运行的进程:"
        ps -u "$username" 2>/dev/null | head -10
        echo
        read -p "是否强制删除并终止所有进程？(y/n): " -n 1 -r
        echo
        if [[ ! $REPLY =~ ^[Yy]$ ]]; then
            log_info "取消删除操作"
            return 0
        fi
        # 杀死所有进程
        pkill -u "$username" 2>/dev/null
        sleep 1
    fi
    
    # 确认删除
    if [ "$remove_home" = "--remove-home" ] || [ "$remove_home" = "-r" ]; then
        read -p "确认删除用户 '$username' 及其家目录？(y/n): " -n 1 -r
        echo
        if [[ ! $REPLY =~ ^[Yy]$ ]]; then
            log_info "取消删除操作"
            return 0
        fi
        
        log_info "正在删除用户及家目录: $username"
        if userdel -r "$username" 2>/dev/null; then
            log_info "✓ 用户 '$username' 及其家目录已删除"
        else
            log_error "删除失败"
            return 1
        fi
    else
        read -p "确认删除用户 '$username'？（家目录将保留）(y/n): " -n 1 -r
        echo
        if [[ ! $REPLY =~ ^[Yy]$ ]]; then
            log_info "取消删除操作"
            return 0
        fi
        
        log_info "正在删除用户: $username"
        if userdel "$username" 2>/dev/null; then
            log_info "✓ 用户 '$username' 已删除（家目录保留在 /home/$username）"
        else
            log_error "删除失败"
            return 1
        fi
    fi
    
    return 0
}

# 修改用户密码
change_password() {
```

```bash
#!/bin/bash

# 获取项目依赖关系
# 用法: ./deps.sh [目录路径]
# 如果不提供路径，则使用当前目录

get_project_dependencies() {
    local target_dir="${1:-.}"  # 如果没有参数，使用当前目录
    
    # 检查目录是否存在
    if [ ! -d "$target_dir" ]; then
        echo "错误: 目录 '$target_dir' 不存在" >&2
        exit 1
    fi
    
    # 转换为绝对路径
    target_dir=$(cd "$target_dir" && pwd)
    
    # 收集所有源文件和头文件（排除隐藏文件）
    files=$(find "$target_dir" -type f \( -name "*.c" -o -name "*.cc" -o -name "*.cpp" -o -name "*.h" -o -name "*.hpp" \) ! -path "*/.*" | sed "s|^$target_dir/||")
    
    # 初始化JSON对象
    echo "{"
    
    local first=true
    while IFS= read -r file; do
        local full_path="$target_dir/$file"
        
        # 获取依赖
        local deps_output
        deps_output=$(g++ -MM "$full_path" 2>/dev/null | sed 's/\\\n//g')
        
        if [ -n "$deps_output" ]; then
            # 提取依赖文件列表（跳过第一个:和第二个元素）
            local deps_list
            deps_list=$(echo "$deps_output" | awk '{for(i=3;i<=NF;i++) print $i}' | while read -r dep; do
                # 转换为相对于目标目录的路径
                local rel_dep
                if [[ "$dep" == /* ]]; then
                    # 绝对路径，检查是否在目标目录内
                    if [[ "$dep" == "$target_dir"/* ]]; then
                        rel_dep="${dep#$target_dir/}"
                        # 检查文件是否存在且不是当前文件
                        if [ -f "$dep" ] && [ "$rel_dep" != "$file" ]; then
                            echo "\"$rel_dep\""
                        fi
                    fi
                else
                    # 相对路径，检查文件是否存在
                    if [ -f "$target_dir/$dep" ] && [ "$dep" != "$file" ]; then
                        echo "\"$dep\""
                    fi
                fi
            done | sort -u | paste -sd ',' -)
            
            # 输出JSON键值对
            if [ "$first" = true ]; then
                first=false
            else
                echo ","
            fi
            
            # 转义文件路径中的特殊字符
            local escaped_file
            escaped_file=$(printf "%s" "$file" | jq -R -s '.')
            
            if [ -n "$deps_list" ]; then
                echo -n "  $escaped_file: [$deps_list]"
            else
                echo -n "  $escaped_file: []"
            fi
        else
            # 处理编译失败的情况
            if [ "$first" = true ]; then
                first=false
            else
                echo ","
            fi
            
            local escaped_file
            escaped_file=$(printf "%s" "$file" | jq -R -s '.')
            echo -n "  $escaped_file: []"
        fi
    done <<< "$files"
    
    echo -e "\n}"
}
```

# 依赖拓扑排序

```bash
#!/bin/bash

# 检查是否提供了 JSON 文件路径
if [ -z "$1" ]; then
    echo "用法: $0 <dependence.json>"
    echo "  结果将直接打印到标准输出，用户可通过 > 重定向保存"
    exit 1
fi

JSON_FILE="$1"

# 检查 jq 是否安装
if ! command -v jq &> /dev/null; then
    echo "错误: 需要安装 jq 来解析 JSON 文件。" >&2
    echo "安装方法: apt-get install jq (Debian/Ubuntu) 或 brew install jq (macOS)" >&2
    exit 1
fi

# 使用临时文件存储节点和依赖关系
NODES_FILE=$(mktemp)
DEPS_FILE=$(mktemp)

# 提取所有节点（文件名）
jq -r 'keys[]' "$JSON_FILE" > "$NODES_FILE"

# 构建依赖关系映射
# 对于每个节点，输出 "节点 依赖节点"（每个依赖一行）
while IFS= read -r node; do
    deps=$(jq -r ".\"$node\"[]" "$JSON_FILE" 2>/dev/null)
    if [ -n "$deps" ]; then
        echo "$deps" | while read -r dep; do
            if [ -n "$dep" ]; then
                echo "$node $dep"
            fi
        done
    fi
done < "$NODES_FILE" > "$DEPS_FILE"

# 计算入度
declare -A indegree
while IFS= read -r node; do
    indegree["$node"]=0
done < "$NODES_FILE"

while read -r node dep; do
    if [ -n "$node" ] && [ -n "$dep" ]; then
        indegree["$node"]=$((indegree["$node"] + 1))
    fi
done < "$DEPS_FILE"

# 初始化队列（入度为0的节点）
declare -a queue
for node in "${!indegree[@]}"; do
    if [ "${indegree[$node]}" -eq 0 ]; then
        queue+=("$node")
    fi
done

# 拓扑排序，逐轮输出 JSONL
round=1
while [ ${#queue[@]} -gt 0 ]; do
    # 将数组转换为 JSON 数组格式
    paths_json=$(printf '%s\n' "${queue[@]}" | jq -R . | jq -s .)
    
    # 输出 JSONL 行到标准输出
    echo "{\"round\": $round, \"paths\": $paths_json}"
    
    # 记录本轮要移除的节点
    declare -A to_remove
    for node in "${queue[@]}"; do
        to_remove["$node"]=1
    done
    
    # 减少依赖这些节点的其他节点的入度
    declare -A new_indegree
    for node in "${!indegree[@]}"; do
        new_indegree["$node"]=${indegree["$node"]}
    done
    
    # 处理依赖关系，减少入度
    while read -r node dep; do
        if [ -n "$node" ] && [ -n "$dep" ] && [ ${to_remove["$dep"]+_} ]; then
            new_indegree["$node"]=$((new_indegree["$node"] - 1))
        fi
    done < "$DEPS_FILE"
    
    # 更新 indegree
    for node in "${!new_indegree[@]}"; do
        indegree["$node"]=${new_indegree["$node"]}
    done
    
    # 重建队列（新的入度为0的节点）
    queue=()
    for node in "${!indegree[@]}"; do
        if [ "${indegree[$node]}" -eq 0 ] && [ -z "${to_remove["$node"]}" ]; then
            queue+=("$node")
        fi
    done
    
    round=$((round + 1))
done

# 清理临时文件
rm -f "$NODES_FILE" "$DEPS_FILE"
```

   ```bash
   #!/usr/bin/env bash
   set -Eeuo pipefail
   
   RED='\033[0;31m'
   GREEN='\033[0;32m'
   YELLOW='\033[1;33m'
   BLUE='\033[0;34m'
   NC='\033[0m'
   
   log_info()    { echo -e "${BLUE}[INFO]${NC} $*"; }
   log_success() { echo -e "${GREEN}[ OK ]${NC} $*"; }
   log_warn()    { echo -e "${YELLOW}[WARN]${NC} $*"; }
   log_error()   { echo -e "${RED}[ERR ]${NC} $*" >&2; }
   
   usage() {
     cat <<'EOF'
   Usage:
     ./run_omx.sh \
       --prompt-template templates/prompt.md \
       --run-template templates/run.yaml \
       --vars vars/task1.yaml \
       --output-dir output \
       [--name task1] \
       [--dry-run]
   
   Options:
     --prompt-template FILE   Prompt 模板文件（{{var}} 占位）
     --run-template FILE      运行参数模板（YAML）
     --vars FILE              变量文件（json/yaml）
     --output-dir DIR         输出目录
     --name NAME              任务名，默认取 vars 文件名
     --dry-run                仅渲染，不执行
     -h, --help               显示帮助
   EOF
   }
   
   require_cmd() {
     command -v "$1" >/dev/null 2>&1 || {
       log_error "缺少命令: $1"
       exit 1
     }
   }
   
   detect_type() {
     local file="$1"
   
     case "${file##*.}" in
       json) echo "json"; return 0 ;;
       yaml|yml) echo "yaml"; return 0 ;;
     esac
   
     if jq empty "$file" >/dev/null 2>&1; then
       echo "json"
       return 0
     fi
   
     if yq -o=json '.' "$file" >/dev/null 2>&1; then
       echo "yaml"
       return 0
     fi
   
     log_error "不支持的文件类型: $file"
     exit 1
   }
   
   to_json() {
     local file="$1"
     local t
     t="$(detect_type "$file")"
   
     if [[ "$t" == "json" ]]; then
       cat "$file"
       return 0
     fi
   
     if ! yq -o=json '.' "$file"; then
       log_error "yq 转换失败: $file"
       exit 1
     fi
   }
   
   escape_sed_replacement() {
     printf '%s' "$1" | sed -e 's/[\/&]/\\&/g'
   }
   
   extract_scalar_kv() {
     local file="$1"
     to_json "$file" | jq -r '
       paths(scalars) as $p
       | [($p | map(tostring) | join(".")), (getpath($p) | tostring)]
       | @tsv
     '
   }
   
   render_template() {
     local template_file="$1"
     local vars_file="$2"
     local out_file="$3"
   
     cp "$template_file" "$out_file"
   
     while IFS=$'\t' read -r key value; do
       [[ -z "$key" ]] && continue
       local escaped
       escaped="$(escape_sed_replacement "$value")"
       sed -i \
         -e "s/{{[[:space:]]*$key[[:space:]]*}}/$escaped/g" \
         "$out_file"
     done < <(extract_scalar_kv "$vars_file")
   }
   
   has_unresolved_placeholders() {
     local file="$1"
     grep -n '{{[[:space:]]*[^}][^}]*}}' "$file" >/dev/null 2>&1
   }
   
   json_get() {
     local json_file="$1"
     local jq_expr="$2"
     jq -r "$jq_expr // empty" "$json_file"
   }
   
   cleanup_files=()
   
   cleanup() {
     local f
     for f in "${cleanup_files[@]:-}"; do
       [[ -n "$f" && -e "$f" ]] && rm -f "$f"
     done
   }
   trap cleanup EXIT
   
   main() {
     local prompt_template=""
     local run_template=""
     local vars_file=""
     local output_dir=""
     local task_name=""
     local dry_run="false"
   
     while [[ $# -gt 0 ]]; do
       case "$1" in
         --prompt-template) prompt_template="$2"; shift 2 ;;
         --run-template)    run_template="$2"; shift 2 ;;
         --vars)            vars_file="$2"; shift 2 ;;
         --output-dir)      output_dir="$2"; shift 2 ;;
         --name)            task_name="$2"; shift 2 ;;
         --dry-run)         dry_run="true"; shift ;;
         -h|--help)         usage; exit 0 ;;
         *)
           log_error "未知参数: $1"
           usage
           exit 1
           ;;
       esac
     done
   
     [[ -n "$prompt_template" ]] || { log_error "缺少 --prompt-template"; exit 1; }
     [[ -n "$run_template" ]]    || { log_error "缺少 --run-template"; exit 1; }
     [[ -n "$vars_file" ]]       || { log_error "缺少 --vars"; exit 1; }
     [[ -n "$output_dir" ]]      || { log_error "缺少 --output-dir"; exit 1; }
   
     [[ -f "$prompt_template" ]] || { log_error "文件不存在: $prompt_template"; exit 1; }
     [[ -f "$run_template" ]]    || { log_error "文件不存在: $run_template"; exit 1; }
     [[ -f "$vars_file" ]]       || { log_error "文件不存在: $vars_file"; exit 1; }
   
     require_cmd jq
   
     local vars_type
     vars_type="$(detect_type "$vars_file")"
     [[ "$vars_type" == "yaml" ]] && require_cmd yq
   
     local run_type
     run_type="$(detect_type "$run_template")"
     [[ "$run_type" == "yaml" ]] && require_cmd yq
   
     OMX_BIN="${OMX_BIN:-$(command -v omx || true)}"
     if [[ -z "$OMX_BIN" ]]; then
       log_error "缺少命令: omx"
       log_error "可通过环境变量 OMX_BIN 指定，例如：OMX_BIN=/home/dev/.local/bin/omx"
       exit 1
     fi
   
     if [[ -z "$task_name" ]]; then
       task_name="$(basename "$vars_file")"
       task_name="${task_name%.*}"
     fi
   
     mkdir -p \
       "$output_dir/prompts" \
       "$output_dir/results" \
       "$output_dir/logs" \
       "$output_dir/meta"
   
     local ts
     ts="$(date +'%Y%m%d_%H%M%S')"
   
     local merged_vars_json vars_json_tmp
     merged_vars_json="$(mktemp /tmp/merged_vars.XXXXXX.json)"
     vars_json_tmp="$(mktemp /tmp/vars_json.XXXXXX.json)"
     cleanup_files+=("$merged_vars_json" "$vars_json_tmp")
   
     to_json "$vars_file" > "$vars_json_tmp"
   
     jq -n \
       --slurpfile vars "$vars_json_tmp" \
       --arg task_name "$task_name" '
       $vars[0] + {task_name: $task_name}
     ' > "$merged_vars_json"
   
     local prompt_rendered="$output_dir/prompts/${task_name}_${ts}.prompt.md"
     local run_rendered="$output_dir/meta/${task_name}_${ts}.run.yaml"
     local run_rendered_json="$output_dir/meta/${task_name}_${ts}.run.json"
     local result_file="$output_dir/results/${task_name}_${ts}.txt"
     local log_file="$output_dir/logs/${task_name}_${ts}.log"
     local meta_file="$output_dir/meta/${task_name}_${ts}.meta.json"
   
     log_info "merged vars:"
     cat "$merged_vars_json"
     echo
   
     render_template "$prompt_template" "$merged_vars_json" "$prompt_rendered"
     render_template "$run_template" "$merged_vars_json" "$run_rendered"
   
     if has_unresolved_placeholders "$prompt_rendered"; then
       log_error "prompt 模板仍有未替换占位符:"
       grep -n '{{[[:space:]]*[^}][^}]*}}' "$prompt_rendered" || true
       exit 1
     fi
   
     if has_unresolved_placeholders "$run_rendered"; then
       log_error "run 模板仍有未替换占位符:"
       grep -n '{{[[:space:]]*[^}][^}]*}}' "$run_rendered" || true
       exit 1
     fi
   
     if ! yq -o=json '.' "$run_rendered" > "$run_rendered_json"; then
       log_error "run 模板渲染后不是合法 YAML: $run_rendered"
       exit 1
     fi
   
     local cd_dir sandbox bypass json_flag output_last_message model profile
     cd_dir="$(json_get "$run_rendered_json" '.cd')"
     sandbox="$(json_get "$run_rendered_json" '.sandbox')"
     bypass="$(json_get "$run_rendered_json" '.bypass')"
     json_flag="$(json_get "$run_rendered_json" '.json')"
     output_last_message="$(json_get "$run_rendered_json" '.output_last_message')"
     model="$(json_get "$run_rendered_json" '.model')"
     profile="$(json_get "$run_rendered_json" '.profile')"
   
     if [[ -n "$output_last_message" ]]; then
       mkdir -p "$(dirname "$output_last_message")"
     fi
   
     local -a cmd
     cmd=("$OMX_BIN" exec -)
   
     [[ -n "$cd_dir" ]] && cmd+=(-C "$cd_dir")
     [[ -n "$sandbox" ]] && cmd+=(-s "$sandbox")
     [[ -n "$output_last_message" ]] && cmd+=(-o "$output_last_message")
     [[ -n "$model" ]] && cmd+=(-m "$model")
     [[ -n "$profile" ]] && cmd+=(-p "$profile")
     [[ "$json_flag" == "true" ]] && cmd+=(--json)
     [[ "$bypass" == "true" ]] && cmd+=(--dangerously-bypass-approvals-and-sandbox)
   
     while IFS= read -r item; do
       [[ -n "$item" ]] && cmd+=(-c "$item")
     done < <(jq -r '.extra_config[]? // empty' "$run_rendered_json")
   
     jq -n \
       --arg task_name "$task_name" \
       --arg prompt_template "$prompt_template" \
       --arg run_template "$run_template" \
       --arg vars_file "$vars_file" \
       --arg prompt_rendered "$prompt_rendered" \
       --arg run_rendered "$run_rendered" \
       --arg result_file "$result_file" \
       --arg log_file "$log_file" \
       --arg ts "$ts" \
       --argjson cmd "$(printf '%s\n' "${cmd[@]}" | jq -R . | jq -s .)" \
       '{
         task_name: $task_name,
         timestamp: $ts,
         prompt_template: $prompt_template,
         run_template: $run_template,
         vars_file: $vars_file,
         prompt_rendered: $prompt_rendered,
         run_rendered: $run_rendered,
         result_file: $result_file,
         log_file: $log_file,
         command: $cmd
       }' > "$meta_file"
   
     log_success "Prompt 已渲染: $prompt_rendered"
     log_success "运行配置已渲染: $run_rendered"
     log_success "元数据已保存: $meta_file"
   
     echo
     log_info "即将执行命令："
     printf ' %q' "${cmd[@]}"
     echo
     echo
   
     if [[ "$dry_run" == "true" ]]; then
       log_warn "dry-run 模式，未执行"
       exit 0
     fi
   
     {
       echo "[COMMAND]"
       printf '%q ' "${cmd[@]}"
       echo
       echo
       echo "[PROMPT]"
       cat "$prompt_rendered"
       echo
       echo "[OUTPUT]"
       cat "$prompt_rendered" | "${cmd[@]}"
     } 2>&1 | tee "$log_file"
   
     log_success "执行完成"
     [[ -n "$output_last_message" ]] && log_success "结果文件: $output_last_message"
     log_success "日志文件: $log_file"
   }
   
   main "$@"
   ```

6. run_all.sh 示例

   ```bash
   #!/usr/bin/env bash
   set -euo pipefail
   
   # 默认值
   DRY_RUN=false
   
   # 解析命令行参数
   while [[ $# -gt 0 ]]; do
       case $1 in
           --dry-run)
               DRY_RUN=true
               shift
               ;;
           --help|-h)
               echo "Usage: $0 [OPTIONS]"
               echo "Options:"
               echo "  --dry-run    Perform a dry run (simulate execution)"
               echo "  --help, -h   Show this help message"
               exit 0
               ;;
   ```

## 专栏文章

1. 在浏览器打开专栏网页，编写脚本获取 URL 链接，然后进行爬取。

```javascript
(function() {
   // 配置参数
   const SCROLL_DELAY_MIN = 3000;  // 最小滚动间隔 3秒
   const SCROLL_DELAY_MAX = 5000;  // 最大滚动间隔 5秒
   const WAIT_AFTER_SCROLL = 1000; // 滚动后等待加载 1秒

   // 存储已抓取的文章（用于去重）
   let fetchedArticles = new Map(); // key: 链接, value: 标题
   let noNewArticleCount = 0;       // 连续无新文章计数
   const MAX_NO_NEW_ROUNDS = 3;      // 连续3轮无新文章则停止

   // 随机延迟函数
   function randomDelay() {
	   const delay = Math.random() * (SCROLL_DELAY_MAX - SCROLL_DELAY_MIN) + SCROLL_DELAY_MIN;
	   return new Promise(resolve => setTimeout(resolve, delay));
   }

   // 延迟函数（固定时间）
   function delay(ms) {
	   return new Promise(resolve => setTimeout(resolve, ms));
   }

   // 滚动到页面底部
   function scrollToBottom() {
	   window.scrollTo({
		   top: document.documentElement.scrollHeight,
		   behavior: 'smooth'
	   });
   }

   // 爬取当前页面的所有微信文章标题和链接
   function scrapeArticles() {
	   const articles = [];
	   // 微信公众平台文章列表常见的选择器
	   const selectors = [
		   '.album__item',
		   '.album__list-item',
		   'li[data-link]',
		   '.js_album_item',
		   '[data-link]'
	   ];

	   // 尝试多个选择器找到文章项
	   let items = [];
	   for (let selector of selectors) {
		   const found = document.querySelectorAll(selector);
		   if (found.length > 0) {
			   items = found;
			   break;
		   }
	   }

	   for (let item of items) {
		   // 获取链接
		   let link = item.getAttribute('data-link') || item.getAttribute('href');
		   if (!link) {
			   const linkEl = item.querySelector('a');
			   if (linkEl) link = linkEl.getAttribute('href');
		   }

		   // 获取标题
		   let title = '';
		   const titleEl = item.querySelector('.album__item-title-wrp, .album__item-title, .weui-mask-ellipsis__text');
		   if (titleEl) {
			   title = titleEl.innerText.trim();
		   } else if (item.innerText) {
			   title = item.innerText.trim().split('\n')[0].slice(0, 100);
		   }

		   // 清理标题中的序号
		   title = title.replace(/^\d+\.\s*/, '');

		   // 过滤有效文章
		   if (link && title && link.startsWith('http') && link.includes('mp.weixin.qq.com')) {
			   articles.push({ title, link });
		   }
	   }

	   // 去重
	   const uniqueArticles = [];
	   const seenLinks = new Set();
	   for (let article of articles) {
		   if (!seenLinks.has(article.link)) {
			   seenLinks.add(article.link);
			   uniqueArticles.push(article);
		   }
	   }

	   return uniqueArticles;
   }

   // 获取新文章
   function getNewArticles(currentArticles) {
	   const newArticles = [];
	   for (let article of currentArticles) {
		   if (!fetchedArticles.has(article.link)) {
			   newArticles.push(article);
			   fetchedArticles.set(article.link, article.title);
		   }
	   }
	   return newArticles;
   }

   // 主循环
   async function main() {
	   console.log('[脚本] 开始抓取微信文章...');
	   
	   while (noNewArticleCount < MAX_NO_NEW_ROUNDS) {
		   await randomDelay();
		   scrollToBottom();
		   await delay(WAIT_AFTER_SCROLL);
		   
		   const currentArticles = scrapeArticles();
		   const newArticles = getNewArticles(currentArticles);
		   
		   if (newArticles.length === 0) {
			   noNewArticleCount++;
		   } else {
			   noNewArticleCount = 0;
		   }
	   }
	   
	   // 最终一次性拼接打印结果（标题\t链接\n）
	   if (fetchedArticles.size > 0) {
		   let result = '';
		   for (let [link, title] of fetchedArticles.entries()) {
			   result += title + '\t' + link + '\n';
		   }
		   console.log(result);
	   } else {
		   console.log('未抓取到任何文章');
	   }
   }
   
   main().catch(console.error);
})();
```

```js
// ==UserScript==
// @name         微信公众号文章批量爬取下载器（基于稳定跳转逻辑）
// @namespace    http://tampermonkey.net/
// @version      2.1
// @description  批量爬取微信公众号文章，自动跳转、提取标题和正文，保存为Markdown文件（含随机延迟防检测）
// @author       Assistant
// @match        https://mp.weixin.qq.com/*
// @match        http://mp.weixin.qq.com/*
// @grant        GM_download
// @grant        GM_setValue
// @grant        GM_getValue
// @grant        GM_deleteValue
// @run-at       document-idle
// ==/UserScript==

(function () {
    'use strict';

    const STATE_KEY = 'wechat_crawl_state';
    const ACTIVE_KEY = 'wechat_crawl_active';

    // 随机延迟配置（毫秒）
    const MIN_DELAY_MS = 2000;  // 最小2秒
    const MAX_DELAY_MS = 4000;  // 最大4秒

    let messageTimer = null;
    let processing = false;

    // ========== 工具函数 ==========
    function getRandomDelay() {
        return Math.floor(Math.random() * (MAX_DELAY_MS - MIN_DELAY_MS + 1)) + MIN_DELAY_MS;
    }

    function randomSleep() {
        const delay = getRandomDelay();
        console.log(`[爬取器] 随机延迟 ${delay}ms 后继续...`);
        return new Promise(resolve => setTimeout(resolve, delay));
    }

    // ========== 状态管理 ==========
    function saveState(state) {
        GM_setValue(STATE_KEY, JSON.stringify(state));
        GM_setValue(ACTIVE_KEY, true);
    }

    function getState() {
        const str = GM_getValue(STATE_KEY, null);
        if (!str) return null;
        try {
            return JSON.parse(str);
        } catch (e) {
            return null;
        }
    }

    function clearState() {
        GM_deleteValue(STATE_KEY);
        GM_deleteValue(ACTIVE_KEY);
    }

    function isActive() {
        return GM_getValue(ACTIVE_KEY, false) === true;
    }

    // ========== URL处理 ==========
    function normalizeInputUrl(url) {
        return url.trim().replace(/&amp;/g, '&');
    }

    function getArticleKey(url) {
        try {
            const u = new URL(url, location.href);
            const path = u.pathname.replace(/\/$/, '');

            if (path.startsWith('/s/') && path.length > 3) {
                return 'short:' + path;
            }

            const biz = u.searchParams.get('__biz') || '';
            const mid = u.searchParams.get('mid') || '';
            const idx = u.searchParams.get('idx') || '';
            const sn = u.searchParams.get('sn') || '';

            if (biz || mid || idx || sn) {
                return `long:${biz}:${mid}:${idx}:${sn}`;
            }

            return 'fallback:' + path;
        } catch (e) {
            return 'raw:' + String(url).split('#')[0].replace(/\/$/, '');
        }
    }

    function isSameArticle(currentUrl, targetUrl) {
        const currentKey = getArticleKey(currentUrl);
        const targetKey = getArticleKey(targetUrl);

        if (currentKey === targetKey) return true;

        try {
            const target = new URL(targetUrl, location.href);
            const current = new URL(currentUrl, location.href);

            if (
                target.pathname.startsWith('/s/') &&
                current.pathname.replace(/\/$/, '') === target.pathname.replace(/\/$/, '')
            ) {
                return true;
            }
        } catch (e) {}

        return false;
    }

    // ========== 内容提取 ==========
    function sanitizeFileName(name) {
        if (!name) return "untitled";
        const illegalChars = /[\\/:*?"<>|]/g;
        let safe = name.replace(illegalChars, '_');
        safe = safe.trim().replace(/^\.+$/, '_').replace(/^\.+|\.+$/g, '');
        if (safe.length === 0) safe = "untitled";
        safe = safe.replace(/[\x00-\x1f\x80-\x9f]/g, '_');
        if (safe.length > 200) safe = safe.slice(0, 200);
        return safe;
    }

    function getArticleTitle() {
        const selectors = [
            '#activity-name',
            'h1.rich_media_title .js_title_inner',
            'h1.rich_media_title',
            '.rich_media_title'
        ];
        for (let sel of selectors) {
            const el = document.querySelector(sel);
            if (el && el.innerText.trim()) {
                return el.innerText.trim();
            }
        }
        let title = document.title;
        if (title && title !== '微信公众平台') {
            return title.replace(/\s*[-|]\s*微信公众平台$/, '').trim();
        }
        return "未获取到标题";
    }

    function getArticleBody() {
        const contentDiv = document.querySelector('.rich_media_content');
        if (!contentDiv) {
            console.warn("未找到 .rich_media_content");
            const fallback = document.querySelector('#js_content');
            if (fallback) return fallback.innerText.trim();
            return "无法提取正文内容";
        }
        const clone = contentDiv.cloneNode(true);
        const removeSelectors = [
            'script', 'style', 'iframe', '.weui-wa-hotarea',
            '.reward_area', '.like_comment', '.profile_container',
            '#js_pc_qr_code', '.qr_code_pc_outer', '.rich_media_extra',
            '.vote_area'
        ];
        removeSelectors.forEach(sel => {
            clone.querySelectorAll(sel).forEach(el => el.remove());
        });
        let text = clone.innerText || "";
        text = text.replace(/\n\s*\n\s*\n/g, '\n\n').trim();
        return text || "正文内容为空";
    }

    async function waitForContent() {
        const maxWaitMs = 15000;
        const intervalMs = 500;
        const start = Date.now();

        while (Date.now() - start < maxWaitMs) {
            const content = document.querySelector('.rich_media_content, #js_content');
            if (content && content.innerText.trim().length > 50) {
                await new Promise(r => setTimeout(r, 500));
                return true;
            }
            await new Promise(r => setTimeout(r, intervalMs));

        return false;
    }

    function downloadMarkdown(content, filename) {
        if (!filename.endsWith('.md')) filename += '.md';
        const blob = new Blob([content], { type: 'text/markdown;charset=utf-8' });
        const url = URL.createObjectURL(blob);

                    URL.revokeObjectURL(url);

                }
            });
        } else {
            fallbackDownload(content, filename);
        }
    }

    function fallbackDownload(content, filename) {
        const blob = new Blob([content], { type: 'text/markdown;charset=utf-8' });
```

   ```python
   def setup_model(
       model_dir="iic/SenseVoiceSmall",
       device="cuda",
   ):
       """初始化 SenseVoiceSmall 模型"""
       print("正在初始化 SenseVoiceSmall 模型...")
   
       if device == "cuda" and not torch.cuda.is_available():
           print("CUDA 不可用，切换到 CPU")

               f"显存: {torch.cuda.get_device_properties(0).total_memory / 1024**3:.1f} GB"
           )
   
       try:
           # 加载 SenseVoiceSmall 模型
           model = AutoModel(
               model=model_dir,
               trust_remote_code=True,
               device=device,
               disable_log=True,  # 减少日志输出

           return model
       except Exception as e:
           print(f"模型加载失败: {e}")

           model = AutoModel(
               model="iic/SenseVoiceSmall",

               device=device,
               disable_log=False,
           )
           return model
   
   
   def transcribe_audio(model, audio_path, output_dir, language="auto", use_itn=True):
       """转录音频文件并保存结果"""
   
       # 创建输出目录
       output_dir = Path(output_dir)
       output_dir.mkdir(exist_ok=True)
   
       # 获取音频文件名（不含扩展名）
       audio_name = Path(audio_path).stem
       output_file = output_dir / f"{audio_name}.txt"
   
       # 检查是否已经转录过（幂等性检查）
       if output_file.exists():
           print(f"跳过 {audio_name} - 转录文件已存在")
           return False
   
       print(f"\n开始转录: {audio_name}")
       start_time = time.time()
   
       try:
           # SenseVoice 支持的语言: auto, zh, en, yue, ja, ko, nospeech
           res = model.generate(
               input=str(audio_path),
               language=language,  # 语言设置
               use_itn=use_itn,  # 是否使用逆文本正则化
               batch_size_s=60,  # 批处理大小（秒）
               merge_vad=True,  # 合并VAD分段
               merge_length_s=15,  # 合并长度
           )
   
           # 后处理结果
           if res and len(res) > 0:
               text = rich_transcription_postprocess(res[0]["text"])
   
               # 保存转录结果
               with open(output_file, "w", encoding="utf-8") as f:
                   f.write(text)
   
               # 实时显示在控制台
               print(f"转录结果: {text}")
   
               # 如果有时间戳信息（可选）
               # SenseVoice 也支持时间戳，需要特殊配置
   
           else:
               raise Exception("未生成转录结果")
   
           elapsed_time = time.time() - start_time
           print(f"转录完成！耗时: {elapsed_time:.1f}秒")
           print(f"结果已保存到: {output_file}")
           return True
   
       except Exception as e:
           print(f"转录 {audio_name} 时出错: {str(e)}")
           # 如果出错，删除可能已创建的临时文件
           if output_file.exists():
               output_file.unlink()
           return False
   
   
   def find_audio_files(input_path):
       """查找所有音频文件"""
       # SenseVoice 主要支持 wav 格式，但也支持其他格式
       audio_extensions = [".wav", ".mp3", ".m4a", ".flac", ".aac", ".ogg"]
   
       if os.path.isfile(input_path):
           # 单个文件
           ext = Path(input_path).suffix.lower()
           if ext in audio_extensions:
               return [Path(input_path)]
           else:
               print(f"不支持的文件格式: {ext}")
               return []
       else:
           # 文件夹
           audio_files = []
           for ext in audio_extensions:
               audio_files.extend(Path(input_path).glob(f"*{ext}"))
               audio_files.extend(Path(input_path).glob(f"*{ext.upper()}"))
           return sorted(audio_files)  # 排序以保持处理顺序一致

       parser.add_argument(
           "--no-itn", action="store_true", help="禁用逆文本正则化（保持原始文本格式）"
       )
       parser.add_argument(
           "--force", "-f", action="store_true", help="强制重新转录已存在的文件"
       )
       parser.add_argument(
           "--device",
           "-d",
           default="cuda",
           choices=["cuda", "cpu"],
           help="运行设备（默认为cuda）",
       )
       parser.add_argument(
           "--model-dir",
           default="/workspace/crawl/projects/audio2text/models/SenseVoiceSmall",
           help="模型目录（默认为iic/SenseVoiceSmall）",
       )
   
       args = parser.parse_args()
   
       # 检查输入路径是否存在
       input_path = Path(args.input)
       if not input_path.exists():
           print(f"错误: 路径不存在 - {input_path}")
           sys.exit(1)
   
       # 查找所有音频文件
       audio_files = find_audio_files(input_path)
   
       if not audio_files:
           print("未找到支持的音频文件")
           print("支持的格式: .wav, .mp3, .m4a, .flac, .aac, .ogg")
           sys.exit(1)
   
       print(f"找到 {len(audio_files)} 个音频文件")
   
       # 设置输出目录
       output_dir = Path(args.output)
       output_dir.mkdir(exist_ok=True)
   
       # 如果需要强制重新转录，删除已存在的输出文件
       if args.force:
           for audio_file in audio_files:
               output_file = output_dir / f"{audio_file.stem}.txt"
               if output_file.exists():
                   output_file.unlink()
                   print(f"删除现有文件: {output_file}")
   
       # 初始化模型
       try:
           model = setup_model(args.model_dir, args.device)
       except Exception as e:
           print(f"模型初始化失败: {e}")
           sys.exit(1)
   
       # 批量处理
       successful = 0
       failed = 0
       skipped = 0
       use_itn = not args.no_itn
   
       for i, audio_file in enumerate(audio_files):
           output_file = output_dir / f"{audio_file.stem}.txt"
   
           # 检查是否已存在（幂等性检查）
           if output_file.exists() and not args.force:
               print(f"跳过 {audio_file.name} - 转录文件已存在")
               skipped += 1
               continue
   
           # 转录音频
           success = transcribe_audio(
               model, audio_file, output_dir, language=args.language, use_itn=use_itn
           )
   
           if success:
               successful += 1
           else:
               failed += 1
   
           # 在批处理之间稍微暂停，避免资源过载
           if len(audio_files) > 1 and i < len(audio_files) - 1:
               time.sleep(0.5)
   
       # 输出统计信息
       print("\n" + "=" * 50)
       print("处理完成！")
       print(f"成功: {successful} 个文件")
       print(f"失败: {failed} 个文件")
       print(f"跳过: {skipped} 个文件")
       print(f"输出目录: {output_dir.absolute()}")
   
       # 性能提示
       # if successful > 0:
       #     print("\n提示: SenseVoiceSmall 推理速度极快")
       #     print("- 约 70ms 处理 10 秒音频")
       #     print("- 比 Whisper-Large 快 15 倍")
   
   
   if __name__ == "__main__":
       main()
   ```

```js
    let collectedUrls = new Set();
    let hasMore = true;
    let scrollAttempts = 0;
    const MAX_SCROLL_ATTEMPTS = 30;

    // 获取当前页面上所有回答的链接
    function getAnswerLinks() {
        const links = [];
        const answerItems = document.querySelectorAll('.ContentItem.AnswerItem');
        
        answerItems.forEach(item => {
            const titleLink = item.querySelector('.ContentItem-title a');
            if (titleLink && titleLink.href) {
                let fullUrl = titleLink.href;
                if (fullUrl.startsWith('//')) {
                    fullUrl = window.location.protocol + fullUrl;
                }
                links.push(fullUrl);
            }
        });
        
        return links;
    }

    // 收集新的链接
    function collectNewUrls() {
        const currentUrls = getAnswerLinks();
        let newCount = 0;
        
        currentUrls.forEach(url => {
            if (!collectedUrls.has(url)) {
                collectedUrls.add(url);
                newCount++;
            }
        });
        
        if (newCount > 0) {
            console.log(`发现 ${newCount} 个新回答链接`);
        }
        
        return newCount;
    }

    // 滚动到页面底部
    function scrollToBottom() {
        return new Promise((resolve) => {
            const maxScroll = document.documentElement.scrollHeight - window.innerHeight;
            
            if (window.scrollY >= maxScroll) {
                resolve(false);
            } else {
                window.scrollTo({
                    top: maxScroll,
                    behavior: 'smooth'
                });
                
                setTimeout(() => {
                    resolve(true);
                }, 1500);
            }
        });
    }

    // 等待新内容加载
    function waitForNewContent(previousCount) {
        return new Promise((resolve) => {
            let checkCount = 0;
            const maxChecks = 15;
            
            const checkInterval = setInterval(() => {
                const currentCount = document.querySelectorAll('.ContentItem.AnswerItem').length;
                
                if (currentCount > previousCount || checkCount >= maxChecks) {
                    clearInterval(checkInterval);
                    resolve(currentCount > previousCount);
                }
                checkCount++;
            }, 500);
        });
    }

    // 主循环
    async function startScraping() {
        console.log('===== 开始采集回答链接 =====');
        
        collectNewUrls();
        
        while (hasMore && scrollAttempts < MAX_SCROLL_ATTEMPTS) {
            const beforeCount = document.querySelectorAll('.ContentItem.AnswerItem').length;
            console.log(`当前已采集 ${collectedUrls.size} 个链接，页面有 ${beforeCount} 个回答项`);
            
            const scrolled = await scrollToBottom();
            
            if (!scrolled) {
                await new Promise(resolve => setTimeout(resolve, 2000));
            }
            
            const hasNewContent = await waitForNewContent(beforeCount);
            const newCount = collectNewUrls();
            
            if (!hasNewContent && newCount === 0) {
                await new Promise(resolve => setTimeout(resolve, 2000));
                const finalCheck = document.querySelectorAll('.ContentItem.AnswerItem').length;
                if (finalCheck === beforeCount) {
                    console.log('没有检测到新内容，采集完成');
                    hasMore = false;
                    break;
                }
            }
            
            scrollAttempts++;
        }
        
        // 最终输出：按换行符拼接后一次性打印
        const allUrls = Array.from(collectedUrls);
        console.log('\n===== 采集完成 =====');
        console.log(`共采集到 ${allUrls.length} 个回答链接：`);
        
        if (allUrls.length > 0) {

    }
```

```js
    function getRandomDelay() {
        return Math.floor(Math.random() * (MAX_DELAY_MS - MIN_DELAY_MS + 1)) + MIN_DELAY_MS;
    }

        const delay = getRandomDelay();
        console.log(`[批量下载器] 随机延迟 ${delay}ms 后继续...`);
        return new Promise(resolve => setTimeout(resolve, delay));
    }

    function normalizeUrl(url) {

    function isSamePage(currentUrl, targetUrl) {
        try {
            const current = new URL(currentUrl);
            const target = new URL(targetUrl);

            // 知乎回答页面：比较 pathname 和 search
            if (current.pathname === target.pathname) return true;

            // 兼容不同参数顺序
            if (current.pathname.includes('/answer/') && target.pathname.includes('/answer/')) {
                const currentMatch = current.pathname.match(/\/answer\/(\d+)/);
                const targetMatch = target.pathname.match(/\/answer\/(\d+)/);
                if (currentMatch && targetMatch && currentMatch[1] === targetMatch[1]) return true;
            }

            // 专栏文章页面：比较 /p/ 后面的数字
            if (current.pathname.includes('/p/') && target.pathname.includes('/p/')) {
                const currentMatch = current.pathname.match(/\/p\/(\d+)/);
                const targetMatch = target.pathname.match(/\/p\/(\d+)/);
                if (currentMatch && targetMatch && currentMatch[1] === targetMatch[1]) return true;
            }

            return false;
        } catch (e) {
            return currentUrl === targetUrl;
        }
    }

    // ========== 内容提取 ==========
    function sanitizeFileName(name) {
        if (!name) return "未命名";
        const illegalChars = /[\\/:*?"<>|]/g;
        let safe = name.replace(illegalChars, '_');
        safe = safe.trim().replace(/^\.+$/, '_').replace(/^\.+|\.+$/g, '');
        if (safe.length === 0) safe = "未命名";
        safe = safe.replace(/[\x00-\x1f\x80-\x9f]/g, '_');
        if (safe.length > 200) safe = safe.slice(0, 200);
        return safe;
    }

    // 提取回答内容

                    const zop = JSON.parse(answerItem.dataset.zop);
                    questionTitle = zop.title || '';

        if (!questionTitle) questionTitle = '未知问题';

        // 获取回答内容
        const contentSelectors = [
            '.RichContent-inner .RichText',
            '.AnswerItem .RichText',
            '.QuestionAnswer-content .RichText',

        let content = '';
        for (let sel of contentSelectors) {
            const el = document.querySelector(sel);
            if (el && el.innerText.trim()) {
                content = el.innerText.trim();

        const authorElem = document.querySelector('.AuthorInfo-name .UserLink-link, .AuthorInfo-name');

        const url = window.location.href;
        const timestamp = new Date().toISOString().slice(0, 19).replace(/:/g, '-');
        const filename = `知乎回答_${sanitizeFileName(questionTitle.substring(0, 50))}_${timestamp}.txt`;
        const textContent = `【问题】${questionTitle}\n【作者】${author}\n【链接】${url}\n【时间】${new Date().toLocaleString()}\n\n${'='.repeat(50)}\n\n${content}`;

    function extractCurrentPage() {
        const url = window.location.href;
        if (url.match(/zhihu\.com\/question\/\d+\/answer\/\d+/)) {
            return extractAnswer();
        } else if (url.match(/zhuanlan\.zhihu\.com\/p\/\d+/)) {
            return extractArticle();
        }
        return null;
    }

    // 等待内容加载完成
    async function waitForContent() {
        const maxWaitMs = 10000;
        const intervalMs = 500;
        const start = Date.now();

        while (Date.now() - start < maxWaitMs) {
            const url = window.location.href;
            if (url.match(/zhihu\.com\/question\/\d+\/answer\/\d+/)) {
                const content = document.querySelector('.RichContent-inner .RichText, .AnswerItem .RichText');
                if (content && content.innerText.trim().length > 20) {
                    await new Promise(r => setTimeout(r, 500));
                    return true;
                }
            } else if (url.match(/zhuanlan\.zhihu\.com\/p\/\d+/)) {
                const content = document.querySelector('.Post-RichText .RichText');
                if (content && content.innerText.trim().length > 50) {
                    await new Promise(r => setTimeout(r, 500));
                    return true;
                }
            }
            await new Promise(r => setTimeout(r, intervalMs));
        }
        return false;
    }

    function downloadContent(filename, content) {
        const blob = new Blob([content], { type: 'text/plain;charset=utf-8' });
        const url = URL.createObjectURL(blob);

        if (typeof GM_download === 'function') {
            GM_download({
                url: url,
                name: filename,
                saveAs: false,
                onload: () => {
                    console.log(`✅ 下载成功: ${filename}`);
                    URL.revokeObjectURL(url);
                },
                onerror: (err) => {
                    console.error(`❌ GM_download失败:`, err);
                    URL.revokeObjectURL(url);
                    fallbackDownload(content, filename);
                }
            });
        } else {
            fallbackDownload(content, filename);
        }
    }

    function fallbackDownload(content, filename) {
        const blob = new Blob([content], { type: 'text/plain;charset=utf-8' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = filename;
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        setTimeout(() => URL.revokeObjectURL(url), 100);
    }

    // ========== 核心：爬取当前页面 ==========
    async function crawlCurrentPage() {
        const state = getState();
        if (!state || !state.active) return false;

        const { urls, currentIndex, total } = state;

        if (!urls || currentIndex >= total) {
            clearState();
            showMessage(`✅ 全部完成！共处理 ${total || 0} 个页面`);
            return false;
        }

        const targetUrl = urls[currentIndex];
        const currentUrl = window.location.href;

        // 检查页面是否匹配
        if (!isSamePage(currentUrl, targetUrl)) {
            console.log('[批量下载器] 当前页不是目标页，跳转中...');
            showMessage(`➡️ 跳转到第 ${currentIndex + 1}/${total} 个页面...`);
            setTimeout(() => {
                window.location.replace(targetUrl);
            }, 500);
            return true;
        }

        console.log(`[批量下载器] [${currentIndex + 1}/${total}] 开始处理: ${targetUrl}`);
        showMessage(`📥 正在处理 (${currentIndex + 1}/${total})...`);

        // 等待内容加载
        const contentReady = await waitForContent();
        if (!contentReady) {
            console.warn('[批量下载器] 内容加载超时，仍尝试提取');
        }

        // 提取内容
        const result = extractCurrentPage();

        if (result && result.textContent) {
            downloadContent(result.filename, result.textContent);
            showMessage(`✅ 已下载: ${result.filename.substring(0, 40)}... (${currentIndex + 1}/${total})`);
            console.log(`[批量下载器] 下载成功: ${result.filename}`);
        } else {
            showMessage(`⚠️ 提取失败: ${targetUrl} (${currentIndex + 1}/${total})`, true);
            console.warn(`[批量下载器] 提取失败: ${targetUrl}`);
        }

        // 等待一下确保下载触发
        await new Promise(r => setTimeout(r, 1000));

        // 更新状态到下一个
        const nextIndex = currentIndex + 1;

        if (nextIndex < total) {
            const newState = {
                urls,
                currentIndex: nextIndex,
                total,
                active: true,
                startTime: state.startTime || Date.now()
            };
            saveState(newState);

            // 随机延迟
            const delayMs = getRandomDelay();
            showMessage(`⏱️ 等待 ${(delayMs / 1000).toFixed(1)} 秒后跳转下一个...`);
            console.log(`[批量下载器] 随机延迟 ${delayMs}ms 后跳转`);
            await randomSleep();

            showMessage(`➡️ 跳转到第 ${nextIndex + 1}/${total} 个页面...`);
            console.log('[批量下载器] 跳转到下一个:', urls[nextIndex]);

            setTimeout(() => {
                window.location.replace(urls[nextIndex]);
            }, 500);
        } else {
            clearState();
            showMessage(`🎉 全部完成！共处理 ${total} 个页面`);
            console.log('[批量下载器] 全部页面处理完成');
        }

        return true;
    }

    async function startTask(urls) {
        if (!urls || urls.length === 0) return false;

        clearState();

        const cleanedUrls = urls.map(normalizeUrl).filter(u => {
            return u.match(/zhihu\.com\/question\/\d+\/answer\/\d+/) ||
                   u.match(/zhuanlan\.zhihu\.com\/p\/\d+/);
        });

        if (cleanedUrls.length === 0) {
            showMessage('❌ 没有有效的知乎回答或专栏文章链接', true);
            return false;
        }

        if (cleanedUrls.length !== urls.length) {
            showMessage(`⚠️ 有 ${urls.length - cleanedUrls.length} 个无效链接被过滤`, true);
        }

        const state = {
            urls: cleanedUrls,
            currentIndex: 0,
            total: cleanedUrls.length,
            active: true,
            startTime: Date.now()
        };

        saveState(state);

        showMessage(`🚀 开始处理，共 ${cleanedUrls.length} 个页面`);
        console.log('[批量下载器] 任务启动:', cleanedUrls);

        setTimeout(() => {
            window.location.replace(cleanedUrls[0]);
        }, 500);

        return true;
    }
```

```js
function parseBody(req) {
  return new Promise((resolve) => {
    const chunks = [];
    req.on("data", (chunk) =>
      chunks.push(Buffer.isBuffer(chunk) ? chunk : Buffer.from(chunk))
    );
    req.on("end", () => {
      const buffer = Buffer.concat(chunks);
      if (!buffer.length) return resolve(null);

      const tryParse = (text) => {
        try { return JSON.parse(text); } catch { return null; }
      };

      let json = tryParse(buffer.toString("utf8"));
      if (json) return resolve(json);

      json = tryParse(iconv.decode(buffer, "gbk"));
      if (json) return resolve(json);

      resolve(null);
    });
  });
}
```

```bash
print_info() { echo -e "${GREEN}[INFO]${NC} $1"; }
print_warn() { echo -e "${YELLOW}[WARN]${NC} $1"; }

  cat <<EOF
用法: $0 [选项] <path>

将普通模型目录转换为 HuggingFace 缓存格式。
修复了旧版本的三个关键 bug:
  1. 使用 HF Hub API 获取正确的 revision SHA 作为 snapshot ID
  2. 对 LFS 文件使用 SHA256，对 git 文件使用 git blob SHA1
  3. 修正符号链接的相对路径计算

参数:
  <path>   模型目录路径（相对路径或绝对路径）

  -s, --size <bytes>    大小阈值（字节），仅当 API 不可用时作为后备，默认 10MB

compute_git_blob_hash() {
  local file="$1"
  git hash-object "$file" 2>/dev/null || {
    # Fallback: compute git blob hash manually
    local size
    size=$(stat -c%s "$file" 2>/dev/null || stat -f%z "$file" 2>/dev/null)
    (printf "blob %s\0" "$size"; cat "$file") | sha1sum | cut -d' ' -f1
  }
}

compute_sha256() {
  sha256sum "$file" 2>/dev/null | cut -d' ' -f1 || shasum -a 256 "$file" 2>/dev/null | cut -d' ' -f1
}

# FIXED: Correct relative path calculation
# Symlink is at:    SNAPSHOT_DIR/<filepath>
# Blob is at:       MODEL_DIR/blobs/<hash>
# From SNAPSHOT_DIR/<dir1>/<file> we need:
#   ../  (for each directory level within snapshot)
#   ../  (to leave SNAPSHOT_DIR)
#   ../  (to leave snapshots/)
#   blobs/<hash>
calculate_relative_path() {
  local file_path="$1"        # e.g., "docs/inference.gif"
  local clean_path="${file_path#./}"

  if [[ "$clean_path" != */* ]]; then
    # File at root of snapshot: need ../../blobs
    echo "../../blobs"
    return
  fi

  # Count directory depth within snapshot
  local depth
  depth=$(echo "$clean_path" | tr -cd '/' | wc -c)
  # Total up-levels = depth + 2 (one for SNAPSHOT_DIR, one for snapshots/)
  local total_up=$((depth + 2))

  local result=""
  for ((i = 0; i < total_up; i++)); do
    result="${result}../"
  done
  echo "${result}blobs"
}
```

```powershell
# ── 6. Hash verification function ──────────────────────────────────────
function Test-FileHashMatch {
    param([string]$Path, $FileInfo)

    $extant = Get-Item $Path -ErrorAction SilentlyContinue
    if (-not $extant -or $extant.Length -eq 0) { return $false }

    # ── SHA256 via LFS OID (model weight files) ──────────────────────
    # This is the ONLY reliable hash check: API's lfs.oid is the true
    # SHA256 of the file content. gitOid is a Git blob OID
    # ("blob <size>\0<content>") — NOT comparable to raw file hash.
    if ($FileInfo.lfsOid) {
        Write-Host "  [VERIFY] $($FileInfo.rfilename) (SHA256...)" -NoNewline
        $actual = (Get-FileHash -Path $Path -Algorithm SHA256).Hash.ToLower()
        $expected = $FileInfo.lfsOid.ToLower()
        if ($actual -eq $expected) {
            Write-Host " OK" -ForegroundColor Green
            return $true
        }
        Write-Host " FAIL" -ForegroundColor Red
        return $false
    }

    # ── Size comparison (non-LFS files) ──────────────────────────────
    # gitOid is NOT a raw file hash (it includes "blob <size>\0" prefix),
    # so we fall back to exact size match instead.
    if ($FileInfo.size -and $extant.Length -eq $FileInfo.size) { return $true }
    if (-not $FileInfo.size) { return $true }   # nothing to compare against → assume OK
    return $false
}
```

```powershell
#!/usr/bin/env pwsh
<#
.SYNOPSIS
  Google Drive -> aria2c URL resolver using curl.exe + jq.
  Extracts direct download URL + filename, prints aria2c command.
.PARAMETER UrlOrId
  GDrive file ID or share URL.
.EXAMPLE
  gdown 1jARfXaU6wiur85Vm1JxZ_xye0FfrUiqb
  gdown "https://drive.google.com/file/d/1jARfXaU6wiur85Vm1JxZ_xye0FfrUiqb/view"
#>
param(
    [Parameter(Position=0)]
    [string]$UrlOrId,
    [Alias("dry")]
    [switch]$InfoOnly
)

# ---- helpers ----
function Get-FileId([string]$s) {
    $m = [regex]::Match($s, '/file/d/([^/]+)')
    if ($m.Success) { return $m.Groups[1].Value }
    $m = [regex]::Match($s, 'id=([^&]+)')
    if ($m.Success) { return $m.Groups[1].Value }
    return $s
}

function Extract-HtmlValue([string]$html, [string]$name) {
    $pat = 'name="' + $name + '"\s+value="([^"]+)"'
    $m = [regex]::Match($html, $pat)
    if ($m.Success) { return $m.Groups[1].Value }
    return ""
}

# ---- resolve ----
if (-not $UrlOrId) { $UrlOrId = Read-Host "Enter GDrive URL or file ID" }
$fileId = Get-FileId $UrlOrId
Write-Host "file_id: $fileId" -ForegroundColor Cyan

$baseUrl = "https://drive.usercontent.google.com/download?id=$fileId&export=download&authuser=0"

# ---- Step 1: confirm page ----
Write-Host "  [1/3] Confirm page ..." -ForegroundColor Cyan
$html = curl.exe -s $baseUrl
$confirm = Extract-HtmlValue $html "confirm"
$uuid    = Extract-HtmlValue $html "uuid"
if (-not $confirm) { $confirm = "t" }
Write-Host "  confirm=$confirm uuid=$uuid"

# ---- Step 2: resolve final URL (fast, no body) ----
Write-Host "  [2/3] Resolve URL ..." -ForegroundColor Cyan
$confirmUrl = "$baseUrl&confirm=$confirm&uuid=$uuid"
$finalUrl = curl.exe -sL -o NUL -w "%{url_effective}" --max-time 8 $confirmUrl 2>&1
if (-not $finalUrl) { $finalUrl = $confirmUrl }

# ---- Step 2b: Get filename (from view page, then CD fallback) ----
Write-Host "  [2b/3] Get filename ..." -ForegroundColor Cyan
$filename = ""

# Try Google Drive view page (most reliable, no download needed)
$viewHtml = curl.exe -sL --max-time 5 "https://drive.google.com/file/d/$fileId/view" 2>&1
$ogMatch = [regex]::Match($viewHtml, '<meta\s+property="og:title"\s+content="([^"]+)"')
if ($ogMatch.Success) {
    $filename = [System.Net.WebUtility]::HtmlDecode($ogMatch.Groups[1].Value)
}
if (-not $filename) {
    $titleMatch = [regex]::Match($viewHtml, '<title>([^<]+)</title>')
    if ($titleMatch.Success) {
        $filename = $titleMatch.Groups[1].Value -replace '\s*[-–—]\s*Google\s+Drive\s*$', ''
    }
}
# Fallback: Content-Disposition from HEAD
if (-not $filename) {
    $headHeaders = curl.exe -sI -L --max-time 8 $finalUrl 2>&1
    $cdLine = ($headHeaders -split "`r?`n") | Where-Object { $_ -match "(?i)Content-Disposition:" } | Select-Object -First 1
    if ($cdLine) {
        $m = [regex]::Match($cdLine, "filename\*=UTF-8''(.+?)(?:;|$)")
        if (-not $m.Success) { $m = [regex]::Match($cdLine, 'filename="(.+?)"') }
        if ($m.Success) { $filename = [System.Net.WebUtility]::UrlDecode($m.Groups[1].Value) }
    }
}
# Final fallback: file ID
if (-not $filename) { $filename = "$fileId" }

# ---- Step 3: output ----
Write-Host "  [3/3] Done" -ForegroundColor Cyan
$result = @{
    url      = $finalUrl
    filename = $filename
    file_id  = $fileId
    aria2c   = 'aria2c -x 8 -s 8 -k 1M --continue=true ' + $finalUrl
}

# Format with jq if available
if (Get-Command jq -ErrorAction SilentlyContinue) {
    $result | ConvertTo-Json -Compress | jq .
} else {
    $result | ConvertTo-Json
}

Write-Host "`n=== Download ===" -ForegroundColor Green
Write-Host $result.aria2c -ForegroundColor Yellow
if ($filename) { Write-Host "  -> $filename" -ForegroundColor Gray }


# ---- Step 4: download ----
if (-not $InfoOnly) {
    Write-Host "`n  [4/3] Downloading ..." -ForegroundColor Cyan
    $aria2cArgs = @(
        "-x", "8"
        "-s", "8"
        "-k", "1M"
        "--continue=true"
        "--console-log-level=notice"
    )
    if ($filename) {
        $aria2cArgs += "--out", $filename
    }
    $aria2cArgs += $finalUrl

    $global:lastexitcode = 0
    & aria2c @aria2cArgs
    if ($global:lastexitcode -ne 0) {
        Write-Host "aria2c failed (exit: $global:lastexitcode). Try downloading manually with the URL above." -ForegroundColor Red
    }
} else {
    Write-Host "`n  [4/3] Skipped (--InfoOnly)" -ForegroundColor Gray
}
```
